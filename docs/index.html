Dockerfile


						FROM php:7.0-apache

						COPY config/php.ini /usr/local/etc/php/
	
						COPY css/ /var/www/html/css/
						COPY js/ /var/www/html/js/
						COPY index.php /var/www/html/
	
						EXPOSE 80
					

						docker run -d -p 80:80 bramdc/docker-demo
